DROP TABLE IF EXISTS soil_data;
DROP TABLE IF EXISTS wifi_info 
DROP TABLE IF EXISTS plant_manage 
DROP TABLE IF EXISTS manage_auto 
DROP TABLE IF EXISTS manage_water 
DROP TABLE IF EXISTS manage_light 

CREATE TABLE IF NOT EXISTS soil_data (id INTEGER PRIMARY KEY, tm REAL, hm REAL, n REAL, p REAL, k REAL, ph REAL, ec INTEGER, lt INTEGER, ts TEXT DEFAULT (datetime('now','localtime')));
CREATE TABLE IF NOT EXISTS wifi_info (id INTEGER PRIMARY KEY, ssid_ap TEXT, pw_ap TEXT, ssid_sta TEXT, pw_sta TEXT);
CREATE TABLE IF NOT EXISTS plant_manage (id INTEGER PRIMARY KEY, w_auto INTEGER, l_auto INTEGER, w_on INTEGER, l_on INTEGER);
CREATE TABLE IF NOT EXISTS manage_auto (id INTEGER PRIMARY KEY, hm REAL, th REAL, lt INTEGER, dr INTEGER);
CREATE TABLE IF NOT EXISTS manage_water (id INTEGER PRIMARY KEY, ud INTEGER, st TEXT, wt INTEGER, no TEXT);
CREATE TABLE IF NOT EXISTS manage_light (id INTEGER PRIMARY KEY, ud INTEGER, st TEXT, ls INTEGER, no TEXT);
